#include<bits/stdc++.h>
using namespace std;

vector<int> adj[1001];
bool visited[1001];

void dfs(int u){
    visited[u] = true;
    for(int v : adj[u]){
        if(!visited[v]){
            dfs(v);
        }
    }
}

int main(){
    int t;
    cin >> t;
    while(t--){
        int c, r;
        cin >> c >> r;
        for(int i=1; i<=c; i++){
            adj[i].clear();
            visited[i] = false;
        }
        for(int i=0; i<r; i++){
            int u, v;
            cin >> u >> v;
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
        int count = 0;
        for(int i=1; i<=c; i++){
            if(!visited[i]){
                dfs(i);
                count++;
            }
        }
        cout << count << endl;
    }
    return 0;
}
